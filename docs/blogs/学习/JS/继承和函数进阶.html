# 继承

  • 遍历父对象的 key,如果子对象中存在 key 就继续遍历，如果不存在就把父对象的属性复制给子对象
var wjl = {
  name: "王健林",
  sex: "男",
  age: "45",
  car: ["玛莎拉蒂", "法拉利"],
  house: ["别墅", "洋房"],

    console.log("高尔夫");
  },
};

var wsc = {
  name: "王思聪",
  sex: "男",
  age: "25",
};
function extend(parent, child) {
  for (var key in parent) {
    if (child[key]) {
      continue;
    }
    child[key] = parent[key];
  }
}

extend(wjl, wsc);

call() 方法使用一个指定的 this 值和单独给出的一个或多个参数来调用一个函数。

该方法的语法和作用与 apply() 方法类似，只有一个区别，就是 call() 方法接受的是一个参数列表，而 apply() 方法接受的是一个包含多个参数的数组。

# arguments

function max() {
  var max = arguments[0];
  for (var i = 0; i < arguments.length; i++) {
    if (max < arguments[i]) {
      max = arguments[i];
    }
  }
  return max;
}

console.log(max(1, 4, 5, 3, 8, 9, 6));

# 冒泡排序

  • 用户设置升序还是降序
Array.prototype.mySort = function(fn) {
  for (var i = 0; i < this, length - 1; i++) {
    var isSort = true; //假设排好序
    for (var j = 0; j < this.length - i - 1; j++) {
      if (fn(this[j], this[j + 1]) > 0) {
        isSort = false;
        var temp = this[j];
        this[j] = this[j + 1];
        this[j + 1] = temp;
      }
    }
    if (isSort) {
      break;
    }
  }
};
var arr = [45, 23, 1, 5, 67, 32, 13, 45, 67, 5, 34, 34, 33, 57, 98];
arr.mySort(function(a, b) {
  return b - a;
});
console.log(arr);

// 闭包特点：延展了函数作用域的范围
function fn() {

  return function() {

  };
}

# 递归

// 求1-n的和
function toAdd(n) {
  if (n === 1) {
    return 1;
  }
  return n + toAdd(n - 1);
}

console.log(toAdd(100));

// 求n的阶乘
function fn(n) {
  if (n === 1) {
    return 1;
  }
  return n * fn(n - 1);
}

console.log(fn(8));

//求斐波那契数列的第n项
function feiBona(n) {
  if (n === 1 || n === 2) {
    return 1;
  }

  return feiBona(n - 1) + feiBona(n - 2);
}

console.log(feiBona(3));

# 对象的拷贝

var Person = {
  name: "蔡文姬",
  age: "20",
  sex: "man",
  play: () => {
    console.log("吃饭");
  },
  dog: {
    name: "金毛",
    age: 2,
  },
  firend: ["张良", "貂蝉"],
};
// 浅拷贝，只复制了第一层，深层的复制的是地址
function copy(parent, child) {
  for (key in parent) {
    child[key] = parent[key];
  }
}

// 深拷贝，深层的也会被拷贝
function deepCopy(parent, child) {
  for (key in parent) {
    var item = parent[key];
    if (item instanceof Object) {
      child[key] = {};
      deepCopy(item, child[key]);
    }
    if (item instanceof Array) {
      child[key] = [];
      deepCopy(item, child[key]);
    } else {
      child[key] = parent[key];
    }
  }
}
var Nong = {};
var Gong = {};
copy(Person, Nong);
// 深层拷贝不会被影响，因为是拷贝的成员，不是地址
deepCopy(Person, Gong);
// 修改parent的成员，不会影响child,因为是复制过去的
Person.sex = "woman";
// 但是如果复制的复杂类型，只复制了地址，会改变
Person.dog.name = "wangwang";
Person.firend[0] = "西施";

console.dir(Nong);
console.dir(Gong);

<h1>递归遍历DOM树</h1>
<p>Tip:可以在遍历的回调函数中任意定制需求</p>
<div>
  <ul id="list">
    <li>射手</li>
    <li>法师</li>
    <li>坦克</li>
    <li>刺客</li>
  </ul>
</div>
<div>
  智商250，魔法，巨大魔法。
  <div>
    鲁班七号
  </div>
</div>
<div>
  垃圾，应该呆在垃圾桶里。
  <div>
    钟馗
  </div>
</div>

<script>
  // 遍历dom树，通过传递一个函数，对相关结点注册事件等操作
  function loadTree(parent, callback) {
    for (var i = 0; i < parent.children.length; i++) {
      var child = parent.children[i];
      // console.log(child);

      if (callback) {
        // 处理找到的子元素
        callback(child);
      }
      // 递归调用
      loadTree(child);
    }
  }

  // loadTree(document.body);

  var ul = document.getElementById("list");
  loadTree(ul, function(element) {
    element.onclick = function() {
      console.log(this.innerText);
    };
  });
</script>
